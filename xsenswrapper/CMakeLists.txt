                                                ${THRIFT_SERVICE_INCLUDE_DIRS})

    target_link_libraries(xsens_mvn_wrapper YARP::YARP_OS YARP::YARP_dev YARP::YARP_sig xsens_mvn_idl yarp_experimental)
